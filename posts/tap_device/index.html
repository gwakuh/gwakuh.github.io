TAP(Terminal Access Point) Device

Kernel Virtual Network Device

물리적인 네트워크 인터페이스가 아닌 가상의 인터페이스를 생성할 때 사용한다.

Send Packet

처음엔 raw socket을 만들어 sendto 함수로 전송하는 로직으로 시도하였으나, TAP Device를 UP 시켜도 LOWER_UP 상태가 아니라 전송이 안되는 것으로 보인다. (추측)

찾아보니 TAP Device를 사용할 때는 송/수신 측을 모두 제어하게 될텐데 뭐하러 소켓 통신을 하냐는 의견을 보고 파일 디스크립터에 직접 write하는 방법을 채용했다.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#include <fcntl.h>
#include <linux/if_tun.h>
#include <linux/limits.h>
#include <net/if.h>
#include <stdint.h>
#include <sys/ioctl.h>
#include <sys/socket.h>

int TapAlloc (char* sDev, int nFlags)
{
    int             nFD;
    struct ifreq    ifr;

    if (sDev == NULL || nFlags < 0)
    {
        return (-1);
    }

    nFD = open ("/dev/net/tun", O_RDWR);
    if (nFD == -1)
    {
        return (-1);
    }

    memset (&ifr, 0, sizeof (ifr));
    ifr.ifr_flags = nFlags;
    strncpy (ifr.ifr_name, sDev, IFNAMSIZ);

    if (ioctl (nFD, TUNSETIFF, &ifr) == -1)
    {
        close (nFD);
        return (-1);
    }

    return (nFD);
}

int TapWrite (int nFD, uint8_t* pPkt, int nPktLen)
{
    if (nFD < 0 || pPkt == NULL || nPktLen < 0)
    {
        return (-1);
    }

    if (write (nFD, pPkt, nPktLen) != nPktLen)
    {
        return (-1);
    }

    return (0);
}

int main ()
{
    int     nFD;
    char    sPkt [] =
        "\x00\x08\x5a\x00\x3b\xbf\x08\x62\x66\x30\x0d\x5e\x08\x00\x45\x00"
        "\x00\x28\x00\x00\x00\x00\xff\x06\x57\xaa\xda\xea\x4d\x30\x79\xa9"
        "\xc2\x61\xc5\x03\x00\x50\xa2\x39\x9a\xbd\x52\x98\x1a\x68\x50\x14"
        "\xff\xff\xdc\x5f\x00\x00";

    nFD = TapAlloc ("tap2", IFF_TAP | IFF_NO_PI);
    if (nFD == -1)
    {
        fprintf (stderr, "Cannot allocate TAP device\n");
        return (-1);
    }

    if (TapWrite (nFD, sPkt, sizeof (sPkt)) == -1)
    {
        fprintf (stderr, "Failed to write packet\n");
        close (nFD);
        return (-1);
    }

    close (nFD);
    return (0);
}

패킷의 형태를 유지하며 전송하기 위해선 이더넷 헤더까지 포함해야 한다.

(예제에선 Ethernet + IP + TCP)
